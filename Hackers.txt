# Common way:

<IfModule mod_rewrite.c>
RewriteCond%{HTTP_USER_AGENT}"(PLACE THE RAW FILE HERE)"[NC]
RewriteRule .* - [F]</ifModule>

#Alternative:
<IfModule mod_rewrite.c>
RewriteCond%{HTTP_USER_AGENT}"(PLACE THE RAW FILE HERE)"[NC]
RewriteRule "^.*$" - [F]</IfModule>

# Extra good BLOCK #: 

<IfModule mod_rewrite.c>
RewriteCond%{HTTP_USER_AGENT}( | | | | )[NC,OR]
RewriteCond%{QUERY_STRING}( | | | )\:[NC,OR]
RewriteCond%{HTTP_USER_AGENT}^.*( | | ).*[NC]
RewriteRule .* - [F]</ifModule>
